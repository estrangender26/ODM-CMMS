4. EQUIPMENT TYPE TO FAMILY MAPPING
------------------------------------
Algorithm: Keyword matching on type_name + type_code + class_name

function getFamilyForEquipmentType(typeName, typeCode, className) {
  const searchText = `${typeName} ${typeCode} ${className}`.toLowerCase();
  
  for (const [familyCode, family] of Object.entries(TemplateFamilies)) {
    for (const keyword of family.equipment_keywords) {
      if (searchText.includes(keyword.toLowerCase())) {
        return familyCode;
      }
    }
  }
  return null;
}

-- Count system templates by family
SELECT 
    SUBSTRING_INDEX(template_code, '_', 1) as family,
    task_kind,
    COUNT(*) as count
FROM task_templates
WHERE is_system = TRUE
GROUP BY family, task_kind
ORDER BY family, task_kind;

-- Count templates with safety controls
SELECT 
    COUNT(DISTINCT tt.id) as templates_with_safety,
    COUNT(tsc.id) as total_safety_controls
FROM task_templates tt
LEFT JOIN task_template_safety_controls tsc ON tt.id = tsc.task_template_id
WHERE tt.is_system = TRUE;

-- Count steps with safety metadata
SELECT 
    COUNT(*) as total_steps,
    SUM(CASE WHEN safety_note IS NOT NULL THEN 1 ELSE 0 END) as steps_with_safety_note,
    SUM(CASE WHEN is_visual_only = TRUE THEN 1 ELSE 0 END) as visual_only_steps,
    SUM(CASE WHEN requires_equipment_stopped = TRUE THEN 1 ELSE 0 END) as stopped_required_steps
FROM task_template_steps s
JOIN task_templates t ON s.task_template_id = t.id
WHERE t.is_system = TRUE;

-- List equipment types with templates
SELECT 
    et.type_name,
    et.type_code,
    COUNT(tt.id) as template_count
FROM equipment_types et
LEFT JOIN task_templates tt ON et.id = tt.equipment_type_id AND tt.is_system = TRUE
GROUP BY et.id
HAVING template_count > 0
ORDER BY template_count DESC;
