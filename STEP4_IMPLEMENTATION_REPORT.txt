================================================================================
VALIDATION INVARIANTS
================================================================================

INVARIANT 1: Family Mapping
- Every equipment_type has exactly one template family mapping
- Check: equipment_type_family_mappings table
- Violation: Missing or duplicate mappings

INVARIANT 2: Industry Mapping  
- Every equipment_type has at least one industry mapping
- Check: equipment_type_industries table
- Violation: No industry associations

INVARIANT 3: System Templates
- Every equipment_type with family mapping has system templates
- Check: task_templates WHERE is_system = TRUE
- Violation: Family mapped but no templates generated

VERIFICATION QUERIES

-- Check family mapping coverage
SELECT 
  COUNT(*) as total,
  SUM(CASE WHEN etm.id IS NOT NULL THEN 1 ELSE 0 END) as mapped
FROM equipment_types et
LEFT JOIN equipment_type_family_mappings etm ON et.id = etm.equipment_type_id;

-- Check industry mapping coverage
SELECT 
  COUNT(*) as total,
  SUM(CASE WHEN eti.id IS NOT NULL THEN 1 ELSE 0 END) as mapped
FROM equipment_types et
LEFT JOIN equipment_type_industries eti ON et.id = eti.equipment_type_id;

-- Check system template coverage
SELECT 
  COUNT(*) as total,
  SUM(CASE WHEN tt.id IS NOT NULL THEN 1 ELSE 0 END) as with_templates
FROM equipment_types et
LEFT JOIN task_templates tt ON et.id = tt.equipment_type_id AND tt.is_system = TRUE;

-- Full coverage check
SELECT 
  COUNT(*) as total_equipment,
  COUNT(CASE WHEN etm.id IS NOT NULL AND eti.id IS NOT NULL AND tt.id IS NOT NULL THEN 1 END) as complete
FROM equipment_types et
LEFT JOIN equipment_type_family_mappings etm ON et.id = etm.equipment_type_id
LEFT JOIN equipment_type_industries eti ON et.id = eti.equipment_type_id
LEFT JOIN task_templates tt ON et.id = tt.equipment_type_id AND tt.is_system = TRUE;
